import { Command } from 'commander';
import { resolve } from 'path';
import { existsSync } from 'fs';
import chalk from 'chalk';
import inquirer from 'inquirer';
import { copyTemplate } from '../utils/file-operations';
import { validatePluginName } from '../utils/validation';

export const createPluginCommand = new Command('create-plugin')
  .description('Create a new Node Lens plugin from template')
  .argument('[name]', 'Plugin name')
  .option('-d, --directory <path>', 'Target directory for the plugin', process.cwd())
  .option('-f, --force', 'Overwrite existing files', false)
  .option('--no-interactive', 'Skip interactive prompts')
  .action(async (pluginName: string | undefined, options) => {
    try {
      let name = pluginName;
      let targetDir = options.directory;

      // Interactive mode for missing information
      if (options.interactive && !name) {
       /* const answers = await inquirer.prompt([
          {
            type: 'input',
            name: 'pluginName',
            message: 'What is the name of your plugin?',
            validate: (input: string) => {
              const validation = validatePluginName(input);
              return validation.valid || validation.message;
            },
          },
          {
            type: 'input',
            name: 'directory',
            message: 'Where would you like to create the plugin?',
            default: process.cwd(),
            when: !targetDir,
          },
        ]);

        name = answers.pluginName;
        if (answers.directory) {
          targetDir = answers.directory;
        }
        */
      }

      if (!name) {
        console.error(chalk.red('Plugin name is required'));
        process.exit(1);
      }

      // Validate plugin name
      const validation = validatePluginName(name);
      if (!validation.valid) {
        console.error(chalk.red(`Invalid plugin name: ${validation.message}`));
        process.exit(1);
      }

      // Resolve target directory
      const finalTargetDir = resolve(targetDir, name);

      // Check if directory exists
      if (existsSync(finalTargetDir) && !options.force) {
        const shouldOverwrite = options.interactive
          ? await inquirer.prompt([
              {
                type: 'confirm',
                name: 'overwrite',
                message: `Directory ${finalTargetDir} already exists. Overwrite?`,
                default: false,
              },
            ])
          : { overwrite: false };

        if (!shouldOverwrite.overwrite) {
          console.log(chalk.yellow('Operation cancelled.'));
          process.exit(0);
        }
      }

      // Copy template
      console.log(chalk.blue(`Creating plugin "${name}" in ${finalTargetDir}...`));

      await copyTemplate(name, finalTargetDir, options.force);

      console.log(chalk.green('✅ Plugin created successfully!'));
      console.log(chalk.cyan(`\\nNext steps:`));
      console.log(chalk.cyan(`  cd ${name}`));
      console.log(chalk.cyan(`  yarn install`));
      console.log(chalk.cyan(`  yarn dev`));

    } catch (error: any) {
      console.error(chalk.red('Failed to create plugin:'), error.message);
      process.exit(1);
    }
  });
